  1. 消息循环

Windows程序是事件驱动的。用户移动鼠标、点击按钮、重绘窗口等都会产生“消息”。消息循环负责不断从系统消息队列中取出消息，翻译（键盘消息），然后分发给对应的窗口过程。

  • WindowClass.lpfnWndProc=MainWindowCallback; 将WindowClass的窗口处理过程绑定程序定义的MainWindowCallback。

  • GetMessage 从消息队列取一条消息（如果队列为空，程序会休眠，不占 CPU）。

  • TranslateMessage 处理按键消息，生成字符消息（例如将 WM_KEYDOWN 转为 WM_CHAR）。

  • DispatchMessage 把消息交给 WindowClass 里注册的窗口过程（即 MainWindowCallback）去处理

也就是说程序定义任何符合规范的函数，绑定到某个类注册的窗口过程，就可以由DispatchMessage调用，这给程序员提供了极大的便利。

callback和表驱动

鉴于程序中大量使用了回调函数的结构，思考是必须的。 回调函数给予了c程序更高的抽象能力，程序将自动执行被注册的回调函数，使程序员只需要关心编写的函数本身。

#include <stdio.h>

// 1. 定义回调函数的类型（类似于 WNDPROC）
typedef int (*MyCallback)(int eventCode, int param);

// 2. 全局变量保存用户注册的回调（模拟每个窗口类绑定的过程）
static MyCallback g_UserCallback = NULL;

// 3. 注册函数（类似于 RegisterClass，把用户的函数指针存起来）
void RegisterMyCallback(MyCallback cb) {
    g_UserCallback = cb;
    printf("[System] Callback registered.\n");
}

// 4. 调度函数（类似于 DispatchMessage）
//    当事件发生时，系统会调用这个函数，它再调用用户注册的回调
void DispatchEvent(int eventCode, int param) {
    if (g_UserCallback != NULL) {
        printf("[System] Event %d occurs, dispatching to user callback...\n", eventCode);
        int result = g_UserCallback(eventCode, param);
        printf("[System] User callback returned %d\n", result);
    } else {
        printf("[System] No callback registered, event ignored.\n");
    }
}

// 5. 用户自定义的“窗口过程”（类似于 MainWindowCallback）
int MyWindowProcedure(int eventCode, int param) {
    printf("[User] MyWindowProcedure called: event=%d, param=%d\n", eventCode, param);
    // 模拟处理不同事件
    switch (eventCode) {
        case 1: return 100;  // 模拟 WM_PAINT 返回
        case 2: return 200;  // 模拟 WM_DESTROY
        default: return 0;
    }
}

int main() {
    // 用户注册自己的回调函数
    RegisterMyCallback(MyWindowProcedure);

    // 模拟系统产生事件，并由 DispatchEvent 调用回调
    DispatchEvent(1, 42);   // 模拟 WM_PAINT
    DispatchEvent(2, 0);    // 模拟 WM_DESTROY

    return 0;
}

//LRESULT : long_ptr
LRESULT CALLBACK MainWindowCallback(HWND Window, UINT Message, WPARAM wParam, LPARAM lParam){
    LRESULT Result=0;
//#if 0 endif 用于禁用代码段，相当于注释

    switch(Message){
        // WM_SIZE 检测是否改变了窗口的大小
        case WM_SIZE:{
            // output to the debugger window
            OutputDebugStringA("WM_SIZE\n");
        }break;
        case WM_DESTROY:{
            OutputDebugStringA("WM_DESTROY\n");
        }break;
        case WM_CLOSE:{
            OutputDebugStringA("WM_CLOSE\n");
        }break;
        // WM_ACTIVEAPP 检测用户是否在使用该窗口
        case WM_ACTIVATEAPP:{
            OutputDebugStringA("WM_ACTIVEAPP\n");
        }break;
        case WM_PAINT:{

            PAINTSTRUCT Paint;
            HDC DeviceContext=BeginPaint(Window,&Paint);
            int X=Paint.rcPaint.left;
            int Y=Paint.rcPaint.top;
            int Height=Paint.rcPaint.bottom-Paint.rcPaint.top;

            int Width=Paint.rcPaint.right-Paint.rcPaint.left;
            static DWORD Operation=WHITENESS;
            PatBlt(DeviceContext,X,Y,Width,Height,Operation);
            if(Operation==WHITENESS){
                Operation=BLACKNESS;
            }
            else{
                Operation=WHITENESS;
            }
            EndPaint(Window,&Paint);

            /*
            HDC BeginPaint(
      [in]  HWND          hWnd,
      [out] LPPAINTSTRUCT lpPaint
    );

            BOOL EndPaint(
              [in] HWND              hWnd,
              [in] const PAINTSTRUCT *lpPaint
            );
            */
        }break;
        default:{
            OutputDebugStringA("default\n");
            Result=DefWindowProc(Window,Message,wParam,lParam);
        }break;
    }

    return(Result);
}

//windows图形界面程序的入口
int CALLBACK WinMain(
    HINSTANCE Instance,
    HINSTANCE PrevInstance,
    LPSTR     CommandLine,
    int       ShowCode
)
{
/*
typedef struct {
    UINT      style;         // 窗口类的样式（CS_HREDRAW, CS_VREDRAW, CS_OWNDC 等）
    WNDPROC   lpfnWndProc;   // 窗口过程函数指针（处理消息）
    int       cbClsExtra;    // 额外类内存（通常为0）
    int       cbWndExtra;    // 额外窗口内存（通常为0）
    HINSTANCE hInstance;     // 应用程序实例句柄（WinMain的第一个参数）
    HICON     hIcon;         // 窗口类的默认图标
    HCURSOR   hCursor;       // 窗口类的默认光标
    HBRUSH    hbrBackground; // 窗口背景画刷
    LPCTSTR   lpszMenuName;  // 菜单资源名称（可以为NULL）
    LPCTSTR   lpszClassName; // 窗口类的名称（字符串）
} WNDCLASS;
*/
    WNDCLASS WindowClass={};
    WindowClass.style=CS_OWNDC|CS_HREDRAW|CS_VREDRAW;
    WindowClass.lpfnWndProc=MainWindowCallback;
    WindowClass.hInstance=Instance;
    // WindowClass.Icon;
    WindowClass.lpszClassName="HandmadeHeroWindowClass";
    //WindowClass.hCursor = LoadCursor(NULL, IDC_ARROW);

    //MessageBox(0,"Gravity Reverse","Warning",MB_OK|MB_ICONINFORMATION);

    if(RegisterClass(&WindowClass)){
        /*
        HWND CreateWindowExW(
  [in]           DWORD     dwExStyle,
  [in, optional] LPCWSTR   lpClassName,
  [in, optional] LPCWSTR   lpWindowName,
  [in]           DWORD     dwStyle,
  [in]           int       X,
  [in]           int       Y,
  [in]           int       nWidth,
  [in]           int       nHeight,
  [in, optional] HWND      hWndParent,
  [in, optional] HMENU     hMenu,
  [in, optional] HINSTANCE hInstance,
  [in, optional] LPVOID    lpParam
);
        */
        HWND WindowHandle=CreateWindowEx(
            0, //扩展窗口样式
            WindowClass.lpszClassName, //窗口类名
            "Handmade Hero", //窗口标题栏文字
            WS_OVERLAPPEDWINDOW|WS_VISIBLE, //窗口样式
            CW_USEDEFAULT, //初始x坐标
            CW_USEDEFAULT, //初始y坐标
            CW_USEDEFAULT, //窗口宽度
            CW_USEDEFAULT, //窗口高度
            0, //父窗口句柄
            0, //菜单句柄
            Instance, //应用程序实例句柄
            0); //创建参数
        if(WindowHandle){
            for(;;){
                MSG Message;
                BOOL MessageResult=GetMessage(&Message,0,0,0);
                if(MessageResult>0){
                    TranslateMessage(&Message);
                    DispatchMessage(&Message);
                }
                else{
                    break;
                }
            }
        }
        else{
        }
    }
    else{
        //todo: logging
    }

    return(0);
}
